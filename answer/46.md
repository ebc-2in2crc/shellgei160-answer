$ cat const26.txt | \
    mecab -E '' | \
    tr ',' ' ' | \
    awk '{print $1, $7=="*" ? $1 : $7}' | \
    sed -E 's/^([^あ-ん]*)([あ-ん]*) (.*)(\2)/\1 \3 \2/' | \
    awk '$1==$2 || NF==1{print $1} $1!=$2 && NF>=2{print $1"("$2")"$3}' | \
    tr -d '\n'
